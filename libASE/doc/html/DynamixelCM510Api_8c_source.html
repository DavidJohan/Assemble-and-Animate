00355 int dynamixelApi_getPositionEstimate(int actuator, bool dir) {
00356         int pos = dynamixelApi_getPosition(actuator);
00357         long currentTime = getLocalMsTime();
00358         int deltaT = currentTime-dyna.estimates[actuator].lastUpdate;
00359         if(dyna.estimates[actuator].estimatonState==0) { //normal mode
00360                 dyna.estimates[actuator].posEstimate = pos;
00361                 if((!dir && pos==1023) || (dir && pos==0)) dyna.estimates[actuator].estimatonState++;
00362         }
00363         else if(dyna.estimates[actuator].estimatonState==1) { //entering blind mode
00364                 dyna.estimates[actuator].posEstimate += ((dir)?-1:1)*deltaT*1024*0.111f*dynamixelApi_getSpeed(actuator)/(60*1000);
00365                 if((!dir && pos==0) || (dir && pos==1023)) dyna.estimates[actuator].estimatonState++;
00366         }
00367         else if(dyna.estimates[actuator].estimatonState==2) { //waiting to leave blind mode
00368                 dyna.estimates[actuator].posEstimate += ((dir)?-1:1)*deltaT*1024*0.111f*dynamixelApi_getSpeed(actuator)/(60*1000);
00369                 if((!dir && pos>0) || (dir && pos<1023)) {
00370                         if(!(pos>100&&pos<(1023-100))) {
00371                                 dyna.estimates[actuator].estimatonState=0;
00372                                 dyna.estimates[actuator].posEstimate = pos;
00373                         }
00374                         else {
00375                                 //printf("spurious reading when crossing center\n");
00376                                 //spurious reading when crossing center
00377                         }
00378                 }
00379         }
00380         dyna.estimates[actuator].lastUpdate = currentTime;
00381         if(dyna.estimates[actuator].posEstimate>(1023+170)) {
00382                 return 1023+170;
00383         }
00384         if(dyna.estimates[actuator].posEstimate<0) {
00385                 if(dyna.estimates[actuator].posEstimate<-170) return 1023;
00386                 else return 1023+170+dyna.estimates[actuator].posEstimate;
00387         }
00388         return dyna.estimates[actuator].posEstimate;
00389 }
