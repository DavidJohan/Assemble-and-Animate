00036         // set UART register A

00047         // set UART register B
00048         // bit7: enable rx interrupt
00049     // bit6: enable tx interrupt

00051     // bit3: enable tx
00052     // bit2: set sendding size(0 = 8bit)
00053         UCSR0B = 0b10011000;
00054         
00055         // set UART register C
00056         // bit6: communication mode (1 = synchronize, 0 = asynchronize)
00057     // bit5,bit4: parity bit(00 = no parity) 
00058     // bit3: stop bit(0 = stop bit 1, 1 = stop bit 2)
00059     // bit2,bit1: data size(11 = 8bit)
00060         UCSR0C = 0b00000110;
00061         
00062         // Set baudrate
00063         Divisor = (unsigned short)(2000000.0 / baudrate) - 1;
00064         UBRR0H = (unsigned char)((Divisor & 0xFF00) >> 8);
00065         UBRR0L = (unsigned char)(Divisor & 0x00FF);
