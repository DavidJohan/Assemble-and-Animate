00036 /*
00037  *
00038  * This example shows how to implement a simple code lock. The code
00039  * lock waits for key presses from a numeric keyboard and if the
00040  * correct code is entered, the lock is unlocked. There is a maximum
00041  * time of one second between each key press, and after the correct
00042  * code has been entered, no more keys must be pressed for 0.5 seconds
00043  * before the lock is opened.
00044  *
00045  * This is an example that shows two things:
00046  * - how to implement a code lock key input mechanism, and
00047  * - how to implement a sequential timed routine.
00048  *
00049  * The program consists of two protothreads, one that implements the
00050  * code lock reader and one that implements simulated keyboard input.
00051  *
00052  *
00053  */
00054

00056 #include <windows.h>
00057 #else
00058 #include <unistd.h>
00059 #include <sys/time.h>
00060 #endif

00093 /*
00094  * The following code implements a simple key input. Input is made
00095  * with the press_key() function, and the function key_pressed()
00096  * checks if a key has been pressed. The variable "key" holds the
00097  * latest key that was pressed. The variable "key_pressed_flag" is set
00098  * when a key is pressed and cleared when a key press is checked.
00099  */
00100 static char key, key_pressed_flag;
00101 
00102 static void
00103 press_key(char k)
00104 {
00105   printf("--- Key '%c' pressed\n", k);
00106   key = k;
00107   key_pressed_flag = 1;
00108 }
00109 
00110 static int
00111 key_pressed(void)
00112 {
00113   if(key_pressed_flag != 0) {
00114     key_pressed_flag = 0;
00115     return 1;
00116   }
00117   return 0;
00118 }

00120 /*
00121  * Declaration of the protothread function implementing the code lock
00122  * logic. The protothread function is declared using the PT_THREAD()
00123  * macro. The function is declared with the "static" keyword since it
00124  * is local to this file. The name of the function is codelock_thread
00125  * and it takes one argument, pt, of the type struct pt.
00126  *
00127  */
00128 static
00129 PT_THREAD(codelock_thread(struct pt *pt))
00130 {
00131   /* This is a local variable that holds the number of keys that have
00132    * been pressed. Note that it is declared with the "static" keyword
00133    * to make sure that the variable is *not* allocated on the stack.
00134    */
00135   static int keys;
00136 
00137   /*
00138    * Declare the beginning of the protothread.
00139    */
00140   PT_BEGIN(pt);
00141 
00142   /*
00143    * We'll let the protothread loop until the protothread is
00144    * expliticly exited with PT_EXIT().
00145    */
00146   while(1) {
00147 
00148     /*
00149      * We'll be reading key presses until we get the right amount of
00150      * correct keys.
00151      */ 
00152     for(keys = 0; keys < sizeof(code); ++keys) {
00153 
00154       /*
00155        * If we haven't gotten any keypresses, we'll simply wait for one.
00156        */
00157       if(keys == 0) {
00158 
00159         /*
00160          * The PT_WAIT_UNTIL() function will block until the condition
00161          * key_pressed() is true.
00162          */
00163         PT_WAIT_UNTIL(pt, key_pressed());
00164       } else {
00165         
00166         /*
00167          * If the "key" variable was larger than zero, we have already
00168          * gotten at least one correct key press. If so, we'll not
00169          * only wait for the next key, but we'll also set a timer that
00170          * expires in one second. This gives the person pressing the
00171          * keys one second to press the next key in the code.
00172          */
00173         timer_set(&codelock_timer, 1000);
00174 
00175         /*
00176          * The following statement shows how complex blocking
00177          * conditions can be easily expressed with protothreads and
00178          * the PT_WAIT_UNTIL() function.
00179          */
00180         PT_WAIT_UNTIL(pt, key_pressed() || timer_expired(&codelock_timer));
00181 
00182         /*
00183          * If the timer expired, we should break out of the for() loop
00184          * and start reading keys from the beginning of the while(1)
00185          * loop instead.
00186          */
00187         if(timer_expired(&codelock_timer)) {
00188           printf("Code lock timer expired.\n");
00189           
00190           /*
00191            * Break out from the for() loop and start from the
00192            * beginning of the while(1) loop.
00193            */
00194           break;
00195         }
00196       }
00197 
00198       /*
00199        * Check if the pressed key was correct.
00200        */
00201       if(key != code[keys]) {
00202         printf("Incorrect key '%c' found\n", key);
00203         /*
00204          * Break out of the for() loop since the key was incorrect.
00205          */
00206         break;
00207       } else {
00208         printf("Correct key '%c' found\n", key);
00209       }
00210     }
00211 
00212     /*
00213      * Check if we have gotten all keys.
00214      */
00215     if(keys == sizeof(code)) {
00216       printf("Correct code entered, waiting for 500 ms before unlocking.\n");
00217 
00218       /*
00219        * Ok, we got the correct code. But to make sure that the code
00220        * was not just a fluke of luck by an intruder, but the correct
00221        * code entered by a person that knows the correct code, we'll
00222        * wait for half a second before opening the lock. If another
00223        * key is pressed during this time, we'll assume that it was a
00224        * fluke of luck that the correct code was entered the first
00225        * time.
00226        */
00227       timer_set(&codelock_timer, 500);      
00228       PT_WAIT_UNTIL(pt, key_pressed() || timer_expired(&codelock_timer));
00229 
00230       /*
00231        * If we continued from the PT_WAIT_UNTIL() statement without
00232        * the timer expired, we don't open the lock.
00233        */
00234       if(!timer_expired(&codelock_timer)) {
00235         printf("Key pressed during final wait, code lock locked again.\n");
00236       } else {
00237 
00238         /*
00239          * If the timer expired, we'll open the lock and exit from the
00240          * protothread.
00241          */
00242         printf("Code lock unlocked.\n");
00243         PT_EXIT(pt);
00244       }
00245     }
00246   }
00247 
00248   /*
00249    * Finally, we'll mark the end of the protothread.
00250    */
00251   PT_END(pt);
00252 }

00353 /*
00354  * This is the main function. It initializes the two protothread
00355  * control structures and schedules the two protothreads. The main
00356  * function returns when the protothread the runs the code lock exits.
00357  */
00358 int
00359 main(void)
00360 {
00361   /*
00362    * Initialize the two protothread control structures.
00363    */
00364   PT_INIT(&input_pt);
00365   PT_INIT(&codelock_pt);
00366 
00367   /*
00368    * Schedule the two protothreads until the codelock_thread() exits.
00369    */
00370   while(PT_SCHEDULE(codelock_thread(&codelock_pt))) {
00371     PT_SCHEDULE(input_thread(&input_pt));
00372     
00373     /*
00374      * When running this example on a multitasking system, we must
00375      * give other processes a chance to run too and therefore we call
00376      * usleep() resp. Sleep() here. On a dedicated embedded system,
00377      * we usually do not need to do this.
00378      */
00379 #ifdef _WIN32
00380     Sleep(0);
00381 #else
00382     usleep(10);
00383 #endif
00384   }
00385 
00386   return 0;

00399 static int clock_time(void)
00400 {
00401   struct timeval tv;
00402   struct timezone tz;   
00403   gettimeofday(&tv, &tz); 
00404   return tv.tv_sec * 1000 + tv.tv_usec / 1000;
00405 }

00409 static int timer_expired(struct timer *t)
00410 { return (int)(clock_time() - t->start) >= (int)t->interval; }
