00045 #define NUM_ITEMS 32
00046 #define BUFSIZE 8
00047 
00048 static int buffer[BUFSIZE];
00049 static int bufptr;
00050 
00051 static void
00052 add_to_buffer(int item)
00053 {
00054   printf("Item %d added to buffer at place %d\n", item, bufptr);  
00055   buffer[bufptr] = item;
00056   bufptr = (bufptr + 1) % BUFSIZE;
00057 }
00058 static int
00059 get_from_buffer(void)
00060 {
00061   int item;
00062   item = buffer[bufptr];
00063   printf("Item %d retrieved from buffer at place %d\n",
00064          item, bufptr);
00065   bufptr = (bufptr + 1) % BUFSIZE;
00066   return item;
00067 }

00069 static int
00070 produce_item(void)
00071 {
00072   static int item = 0;
00073   printf("Item %d produced\n", item);
00074   return item++;
00075 }
00076 
00077 static void
00078 consume_item(int item)
00079 {
00080   printf("Item %d consumed\n", item);
00081 }

00083 static struct pt_sem full, empty;
00084  
00085 static 
00086 PT_THREAD(producer(struct pt *pt))
00087 {
00088   static int produced;
00089   
00090   PT_BEGIN(pt);
00091   
00092   for(produced = 0; produced < NUM_ITEMS; ++produced) {
00093   
00094     PT_SEM_WAIT(pt, &full);
00095     
00096     add_to_buffer(produce_item());
00097     
00098     PT_SEM_SIGNAL(pt, &empty);
00099   }
00100 
00101   PT_END(pt);
00102 }
00103  
00104 static 
00105 PT_THREAD(consumer(struct pt *pt))
00106 {
00107   static int consumed;
00108   
00109   PT_BEGIN(pt);
00110  
00111   for(consumed = 0; consumed < NUM_ITEMS; ++consumed) {
00112     
00113     PT_SEM_WAIT(pt, &empty);
00114     
00115     consume_item(get_from_buffer());    
00116     
00117     PT_SEM_SIGNAL(pt, &full);
00118   }
00119  
00120   PT_END(pt);
00121 }

00123 static 
00124 PT_THREAD(driver_thread(struct pt *pt))
00125 {
00126   static struct pt pt_producer, pt_consumer;
00127  
00128   PT_BEGIN(pt);
00129   
00130   PT_SEM_INIT(&empty, 0);
00131   PT_SEM_INIT(&full, BUFSIZE);
00132  
00133   PT_INIT(&pt_producer);
00134   PT_INIT(&pt_consumer);
00135  
00136   PT_WAIT_THREAD(pt, producer(&pt_producer) &
00137                      consumer(&pt_consumer));
00138  
00139   PT_END(pt);
00140 }
