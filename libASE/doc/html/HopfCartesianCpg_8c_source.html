00069 void HCCpg_iterate(HCCpg_t *cpg) {
00070         if(Strategy_isPaused((Strategy_t*)cpg) || !Strategy_isStarted((Strategy_t*)cpg)) return;
00071         if(cpg->parent.nCouplings!=0) cpg->omega_add=cpg->omega_add/cpg->parent.nCouplings;
00072         float x_next = RK4_itterate(0, cpg->parent.x, cpg->h, cpg, f1); //time=0 since f1 and f2 are independent from t
00073         float y_next = RK4_itterate(0, cpg->parent.y, cpg->h, cpg, f2);
00074         cpg->parent.x = x_next;
00075         cpg->parent.y = y_next;
00076 
00077         if(fabs(cpg->parent.x)>5||fabs(cpg->parent.y)>5)  {
00078                 #ifdef PRINTF

00080                 #endif
00081                 HCCpg_resetState(cpg);
00082         }
00083 
00084         float p = cpg->h/6.283f;
00085         cpg->omegaEffective =(1-p)*cpg->omegaEffective + p*(cpg->omega+cpg->omega_add);

00087         cpg->y_add = 0;
